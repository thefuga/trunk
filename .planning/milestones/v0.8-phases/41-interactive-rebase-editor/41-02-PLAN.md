    **Drag-and-drop (native HTML5 DnD):**

    Add these handler functions (if not already present from Task 1's basic scaffold):

    ```typescript
    function handleDragStart(e: DragEvent, idx: number) {
      dragIdx = idx;
      e.dataTransfer?.setData('text/plain', String(idx));
      if (e.dataTransfer) e.dataTransfer.effectAllowed = 'move';
    }

    function handleDragOver(e: DragEvent, idx: number) {
      e.preventDefault();
      if (dragIdx === null || dragIdx === idx) return;
      // Swap: remove from old position, insert at new
      const updated = [...items];
      const [moved] = updated.splice(dragIdx, 1);
      updated.splice(idx, 0, moved);
      items = updated;
      // Update focusedIndex to follow the moved item
      focusedIndex = idx;
      dragIdx = idx;
    }

    function handleDragEnd() {
      dragIdx = null;
    }
    ```

    Wire these to row div attributes:
    - `ondragstart={(e) => handleDragStart(e, idx)}`
    - `ondragover={(e) => handleDragOver(e, idx)}`
    - `ondragend={handleDragEnd}`

    **Keyboard shortcuts:**

    Add a keydown handler on the editor container div (not window, to scope shortcuts):
    ```typescript
    function handleEditorKeydown(e: KeyboardEvent) {
      // Don't handle shortcuts when focus is inside a select, input, or textarea
      const tag = (e.target as HTMLElement)?.tagName;
      if (tag === 'SELECT' || tag === 'INPUT' || tag === 'TEXTAREA') return;

      switch (e.key) {
        case 'p':
        case 'P':
          e.preventDefault();
          items[focusedIndex].action = 'pick';
          break;
        case 's':
        case 'S':
          e.preventDefault();
          items[focusedIndex].action = 'squash';
          break;
        case 'r':
        case 'R':
          e.preventDefault();
          items[focusedIndex].action = 'reword';
          break;
        case 'd':
        case 'D':
          e.preventDefault();
          items[focusedIndex].action = 'drop';
          break;
        case 'ArrowUp':
          e.preventDefault();
          focusedIndex = Math.max(0, focusedIndex - 1);
          scrollRowIntoView(focusedIndex);
          break;
        case 'ArrowDown':
          e.preventDefault();
          focusedIndex = Math.min(items.length - 1, focusedIndex + 1);
          scrollRowIntoView(focusedIndex);
          break;
        case 'Escape':
          e.preventDefault();
          onclose();
          break;
      }
    }
    ```

    Add `scrollRowIntoView` helper:
    ```typescript
    function scrollRowIntoView(idx: number) {
      const row = document.querySelector(`[data-rebase-row="${idx}"]`);
      row?.scrollIntoView({ block: 'nearest' });
    }
    ```

    Wire to the editor's outermost container div:
    - `onkeydown={handleEditorKeydown}`
    - `tabindex="-1"` to make it focusable
    - Add `use:autofocus` action to auto-focus on mount (function: `function autofocus(node: HTMLElement) { node.focus(); }`)

    Add `data-rebase-row={idx}` attribute to each row div for scrollRowIntoView targeting.

    Ensure the row click handler sets `focusedIndex = idx` and does NOT steal focus from the container (the click should not focus the select dropdown, just highlight the row).
