```typescript
async function showOverflowRefContextMenu(e: MouseEvent, ref: RefLabel) {
  e.preventDefault();
  e.stopPropagation();

  if (ref.ref_type === 'LocalBranch') {
    const menu = await Menu.new({
      items: [
        await MenuItem.new({
          text: 'RenameŌĆ”',
          action: () => { handleRenameBranch(ref.short_name); },
        }),
        await PredefinedMenuItem.new({ item: 'Separator' }),
        await MenuItem.new({
          text: 'Delete',
          enabled: !ref.is_head,
          action: () => { handleDeleteBranch(ref.short_name).catch(() => {}); },
        }),
      ],
    });
    await menu.popup();
  } else if (ref.ref_type === 'Tag') {
    const menu = await Menu.new({
      items: [
        await MenuItem.new({
          text: 'Delete',
          action: () => { handleDeleteTag(ref.short_name).catch(() => {}); },
        }),
      ],
    });
    await menu.popup();
  }
}
```

Note: This mirrors `showPillContextMenu` but reads from `RefLabel` fields (`ref_type`, `short_name`, `is_head`) instead of `OverlayRefPill` fields (`refType`, `label`, `isHead`). A thin wrapper is cleaner than refactoring `showPillContextMenu` to accept a union type ŌĆö avoids touching tested code paths.
