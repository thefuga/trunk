function isRemoteOnly(ref: RefLabel): boolean {
  if (ref.ref_type !== 'RemoteBranch') return false;
  return !refs.some(
    (r) => r !== ref && (r.ref_type === 'LocalBranch' || r.ref_type === 'Tag')
  );
}
