```typescript
/**
 * Count the total number of files (recursively) within a tree node array.
 * Counts only file nodes, not directory nodes.
 */
export function countFiles(nodes: TreeNode[]): number {
  let count = 0;
  for (const node of nodes) {
    if (node.type === 'file') {
      count++;
    } else {
      count += countFiles(node.children);
    }
  }
  return count;
}
```

**B. Add collectFilePaths utility to build-tree.ts (TREE-08 support):**

Add another exported function for collecting all file paths under a directory:

```typescript
/**
 * Collect all file paths recursively from a tree node array.
 */
export function collectFilePaths(nodes: TreeNode[]): string[] {
  const paths: string[] = [];
  for (const node of nodes) {
    if (node.type === 'file') {
      paths.push(node.path);
    } else {
      paths.push(...collectFilePaths(node.children));
    }
  }
  return paths;
}
```

**C. Add directory staging functions (TREE-08, per D-10, D-11):**

```typescript
async function stageDirectory(dirPath: string) {
  // Build tree from unstaged files and find all file paths under dirPath
  const tree = buildTree(status?.unstaged ?? []);
  const allPaths = collectFilePaths(tree);
  const matchingPaths = allPaths.filter(p => p.startsWith(dirPath + '/') || p === dirPath);
  // Also include files directly under the directory prefix in the flat list
  const directMatches = (status?.unstaged ?? []).filter(f => f.path.startsWith(dirPath + '/') || f.path.startsWith(dirPath));
  const pathsToStage = [...new Set([...matchingPaths, ...directMatches.map(f => f.path)])];

  // Stage all matching files in parallel (per Claude's discretion: Promise.all for performance)
  loadingFiles = new Set([...loadingFiles, ...pathsToStage]);
  await Promise.all(pathsToStage.map(p => safeInvoke('stage_file', { path: repoPath, filePath: p })));
  await loadStatus();
  const next = new Set(loadingFiles);
  for (const p of pathsToStage) next.delete(p);
  loadingFiles = next;
}

async function unstageDirectory(dirPath: string) {
  const directMatches = (status?.staged ?? []).filter(f => f.path.startsWith(dirPath + '/') || f.path.startsWith(dirPath));
  const pathsToUnstage = directMatches.map(f => f.path);

  loadingFiles = new Set([...loadingFiles, ...pathsToUnstage]);
  await Promise.all(pathsToUnstage.map(p => safeInvoke('unstage_file', { path: repoPath, filePath: p })));
  await loadStatus();
  const next = new Set(loadingFiles);
  for (const p of pathsToUnstage) next.delete(p);
  loadingFiles = next;
}
```
